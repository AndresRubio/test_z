# Web UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A brand-styled, single-page web UI served by the existing FastAPI app at `GET /`, acting as a pure client of `POST /chat` and `GET /health`.

**Architecture:** One self-contained `app/ui/static/index.html` (inline CSS + JS, no build step) served by a tiny router in `app/ui/__init__.py`, included by `create_app`. Zero backend/pipeline changes; zero new dependencies.

**Tech Stack:** FastAPI (existing), vanilla HTML/CSS/JS, pytest + httpx `ASGITransport` (existing test pattern).

**Spec:** `docs/specs/web-ui/2026-07-10-web-ui-design.md`

## Global Constraints

- **`uv` + Python 3.12 only.** All commands via `uv run …`. Never `pip install`.
- **The client's brand name must never appear in any file** — code, comments,
  HTML, commit messages. The palette below is expressed as hex values under
  neutral names; keep it that way. No logo, no brand assets, no brand URLs.
- **No new dependencies.** `pyproject.toml` and `uv.lock` must be untouched.
- **No changes to `app/api/`, `app/chat/`, `app/retrieval/`, `app/llm/`,
  `app/catalog/`, or any existing test.** This feature only adds files plus
  two include-lines in `app/main.py` and short doc notes.
- Tests are offline (no Ollama, no network), `asyncio_mode = "auto"` — write
  bare `async def test_…` functions, no decorators.
- Lint gate: `uv run ruff check app tests evals` must stay clean.

```python
"""Static web UI for manual testing and demos — a pure client of the API."""

from pathlib import Path

from fastapi import APIRouter
from fastapi.responses import FileResponse

_STATIC_DIR = Path(__file__).parent / "static"

router = APIRouter()


@router.get("/", include_in_schema=False)
async def index() -> FileResponse:
    return FileResponse(_STATIC_DIR / "index.html", media_type="text/html")
```

```html
  const siteSelect = document.getElementById("site-select");
  const statusPill = document.getElementById("status-pill");
  const banner = document.getElementById("banner");
  const transcript = document.getElementById("transcript");
  const composer = document.getElementById("composer");

  function setBusy(busy) {
    queryInput.disabled = busy;
    sendBtn.disabled = busy;
    siteSelect.disabled = busy;

  function addBubble(cls, node) {
    const bubble = document.createElement("div");
    bubble.className = "bubble " + cls;
    bubble.append(node);
    transcript.append(bubble);
    bubble.scrollIntoView({ block: "end" });
    return bubble;
  }

  function textNode(text) {
    return document.createTextNode(text);
  }

  function formatPrice(price, currency) {
    try {
      return new Intl.NumberFormat(undefined, { style: "currency", currency }).format(price);
    } catch {
      return price + " " + currency;
    }
  }

  function renderCard(card) {
    const el = document.createElement("div");
    el.className = "card";

    const name = document.createElement("div");
    name.className = "name";
    name.textContent = card.product_name;
    el.append(name);

    const price = document.createElement("div");
    price.className = "price";
    price.textContent = formatPrice(card.price, card.currency);
    el.append(price);

      const stars = document.createElement("div");
      stars.className = "stars";
      stars.textContent =
        "★ " + card.rating_average.toFixed(1) + " (" + card.rating_count + ")";
      el.append(stars);
    }

    badges.className = "badges";
    const pet = document.createElement("span");
    pet.className = "badge pet";
    pet.textContent = card.pet_type;
    badges.append(pet);
    if (card.discount_label) {
      const discount = document.createElement("span");
      discount.className = "badge discount";
      discount.textContent = card.discount_label;
      badges.append(discount);
    }
    const stock = document.createElement("span");
    stock.className = "badge " + (card.in_stock ? "stock-in" : "stock-out");
    stock.textContent = card.in_stock ? "In stock" : "Out of stock";
    badges.append(stock);
    el.append(badges);

    return el;
  }

  function renderAnswer(siteId, body) {
    const wrap = document.createElement("div");

    const meta = document.createElement("div");
    meta.className = "meta";
    meta.textContent =
      "Site " + siteId + " · " + body.retrieved_products.count + " product(s) retrieved";
    wrap.append(meta);

    wrap.append(textNode(body.answer));

    if (body.retrieved_products.count > 0) {
      const cards = document.createElement("div");
      cards.className = "cards";
      for (const card of body.retrieved_products.products) {
        cards.append(renderCard(card));
      }
      wrap.append(cards);

    raw.className = "raw";
    const summary = document.createElement("summary");
    summary.textContent = "raw JSON";

    pre.textContent = JSON.stringify(body, null, 2);
    raw.append(summary, pre);

  async function loadHealth() {
    banner.hidden = true;
    statusPill.textContent = "checking…";
    statusPill.className = "pill";
    try {
      const response = await fetch("/health");
      if (!response.ok) throw new Error("health " + response.status);
      const health = await response.json();
      siteSelect.replaceChildren(
        ...health.sites.map((siteId) => new Option("Site " + siteId, siteId))
      );
      const reachable = health.ollama === "reachable";

      statusPill.className = "pill bad";
      banner.hidden = false;
    }

    queryInput.value = "";

    try {
      const response = await fetch("/chat", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ site_id: siteId, query }),

      typing.remove();

        renderAnswer(siteId, await response.json());

        let detail = "HTTP " + response.status;

          detail = (await response.json()).detail || detail;
        } catch {}
        addBubble("warning", textNode(detail));
      }
    } catch {
      typing.remove();
      addBubble("warning", textNode("Network error — server unreachable. Check the header status."));
      loadHealth();
    } finally {
      setBusy(false);
    }
  });
```
